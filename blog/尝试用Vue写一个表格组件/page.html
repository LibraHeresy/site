<script>
import { computed, onMounted } from "vue";
import Taro from "@tarojs/taro";

export default {
  props: {
    columns: {
      type: Array,
      default: () => [],
    },
    dataSource: {
      type: Array,
      default: () => [],
    },
    showEmpty: {
      type: Boolean,
      default: true,
    },

      type: Boolean,

  setup(props) {

        "grid-template-columns": setWidth(),

      return props.columns

          return item.width ? item.width : "1fr";

        .join(" ");
    };

    return {
      colStyle,

  },
};
